func getMax(arr []int, l, r int) int {
	// base case
	if l == r {
		return arr[l]
	}
	// l<r ,m是中点
	// l...m......r
	//l...m
	//      m+1...r
	m := (l + r) / 2
	leftMax := getMax(arr, l, m)
	rightMax := getMax(arr, m+1, r)

	if leftMax >= rightMax {
		return leftMax
	} else {
		return rightMax
	}
}
